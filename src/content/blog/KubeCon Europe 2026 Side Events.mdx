(function() {
  function init() {
    const dayData = {
      '18': { label: 'Wednesday — Pre-Week', date: '18 March' },
      '21': { label: 'Saturday — Rejekts', date: '21 March' },
      '22': { label: 'Sunday — KubeTrain', date: '22 March' },
      '23': { label: 'Monday — Co-Located', date: '23 March' },
      '24': { label: 'Tuesday — KubeCon Day 1', date: '24 March' },
      '25': { label: 'Wednesday — KubeCon Day 2', date: '25 March' },
      '26': { label: 'Thursday — KubeCon Day 3', date: '26 March' },
    };
    const floatingDay = document.getElementById('floating-day');
    const floatingLabel = document.getElementById('floating-label');
    const floatingDate = document.getElementById('floating-date');
    const pills = document.querySelectorAll('.day-pill');
    if (!floatingDay || !floatingLabel || !floatingDate) return;
    let currentDay = null;
    let scrollTimeout;
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const day = entry.target.getAttribute('data-day');
          if (day && day !== currentDay) {
            currentDay = day;
            if (dayData[day]) {
              floatingLabel.textContent = dayData[day].label;
              floatingDate.textContent = dayData[day].date;
              floatingDay.style.opacity = '1';
              floatingDay.style.transform = 'translateY(0)';
            }
            pills.forEach(p => p.classList.toggle('active', p.getAttribute('data-day') === day));
            const header = entry.target.querySelector('.day-header');
            if (header) {
              header.classList.remove('in-view');
              void header.offsetWidth;
              header.classList.add('in-view');
            }
          }
        }
      });
    }, { rootMargin: '-10% 0px -40% 0px' });
    const sections = document.querySelectorAll('.day-section');
    sections.forEach(el => observer.observe(el));
    window.addEventListener('scroll', () => {
      floatingDay.style.opacity = '1';
      floatingDay.style.transform = 'translateY(0)';
      clearTimeout(scrollTimeout);
      scrollTimeout = setTimeout(() => {
        floatingDay.style.opacity = '0';
        floatingDay.style.transform = 'translateY(16px)';
      }, 2000);
    });
    pills.forEach(pill => {
      pill.addEventListener('click', (e) => {
        e.preventDefault();
        const target = document.querySelector(pill.getAttribute('href'));
        if (target) target.scrollIntoView({ behavior: 'smooth', block: 'start' });
      });
    });
  }
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
`}</script>
